CREATE TABLE isla(
  codigo NUMERIC(10,0) PRIMARY KEY,
  nombre VARCHAR(50) NOT NULL UNIQUE,
  region VARCHAR(20) NOT NULL,
  fruta_diablo NUMERIC(10,0) UNIQUE REFERENCES fruta_del_diablo
);

CREATE TABLE enfrentamiento(
  numero NUMERIC(10,0) PRIMARY KEY, 
  nombre VARCHAR(100) NOT NULL UNIQUE,
  fecha DATE NOT NULL,
  numero_bajas NUMERIC(10,0) NOT NULL,
  lugar_inicio NUMERIC(10,0) NOT NULL REFERENCES isla,
  lugar_fin NUMERIC(10,0) REFERENCES isla
);

CREATE TABLE fruta_del_diablo(
  codigo NUMERIC(10,0) PRIMARY KEY,
  nombre VARCHAR(50) NOT NULL UNIQUE,
  tipo VARCHAR(20) NOT NULL,
  poder_otorgado VARCHAR(100) NOT NULL UNIQUE,
  precio NUMERIC(10,0) NOT NULL,
  fecha_produccion DATE NOT NULL,
  fecha_expiracion DATE NOT NULL,
);
